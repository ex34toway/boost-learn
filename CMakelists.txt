cmake_minimum_required(VERSION 3.14)

project (boost-learn)

set(CMAKE_CXX_STANDARD 11)

option(ENABLE_TEST "Build all tests." ON)

set(Boost_USE_STATIC_LIBS ON)
find_package(Boost REQUIRED COMPONENTS regex thread)

include_directories(${Boost_INCLUDE_DIRS})

if (ENABLE_TEST)
include(FetchContent)
FetchContent_Declare(
  googletest
  URL https://github.com/google/googletest/archive/refs/tags/release-1.11.0.zip
)

set(gtest_force_shared_crt ON CACHE BOOL "" FORCE)
FetchContent_MakeAvailable(googletest)

enable_testing()
add_executable(signals2_test tests/signals2/signals2_test.cpp)
target_link_libraries(signals2_test PRIVATE gtest_main)

add_executable(asio_test tests/asio/asio_timer_test.cpp tests/asio/asio_udp_test.cpp tests/asio/asio_tcp_test.cpp)
target_link_libraries(asio_test PRIVATE gtest_main ${Boost_LIBRARIES})

include(GoogleTest)
gtest_discover_tests(signals2_test)
gtest_discover_tests(asio_test)
endif()
